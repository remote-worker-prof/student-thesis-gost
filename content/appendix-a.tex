def update_priority(skill_score: float, fatigue: float) -> float:
    base = 0.7 * (1.0 - skill_score)
    penalty = 0.3 * fatigue
    return max(0.0, min(1.0, base + penalty))
\end{lstlisting}

pub fn moving_average(scores: &[f64], window: usize) -> Vec<f64> {
    scores
        .windows(window)
        .map(|chunk| chunk.iter().sum::<f64>() / window as f64)
        .collect()
}
\end{lstlisting}

\section*{Python: постобработка обратной связи}

\begin{lstlisting}[style=sthgcode-python,caption={Пример обновления профиля после экспертной ревизии}]
def apply_expert_feedback(profile: dict, delta_reward: float) -> dict:
    profile = dict(profile)
    profile["reward_bias"] = round(profile.get("reward_bias", 0.0) + delta_reward, 4)
    profile["manual_reviews"] = int(profile.get("manual_reviews", 0)) + 1
    profile["needs_retrain"] = abs(profile["reward_bias"]) > 0.25
    return profile
\end{lstlisting}
